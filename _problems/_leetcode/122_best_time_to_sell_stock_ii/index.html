    public int maxProfit(int[] prices) {
    int N = prices.length;
    int dp0 = 0, dp1 = -prices[0];
    for (int i = 1; i < N; ++i) {
        dp0 = Math.max(dp0, dp1 + prices[i]);
        dp1 = Math.max(dp1, dp0 - prices[i]);
    }
    return dp0;
}

    public int maxProfit(int[] prices) {
    int profits = 0;
    for (int i = 1; i < prices.length; ++i) {
        if (prices[i] > prices[i - 1])
            profits += prices[i] - prices[i - 1];
    }
    return profits;
}
